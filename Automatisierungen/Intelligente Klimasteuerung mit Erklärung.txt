automation:
  - id: 'smart_climate_adjustment'
    alias: "Intelligente Klimaanpassung"
    trigger:
      - platform: numeric_state
        entity_id: sensor.outdoor_temperature
        above: 25
    condition:
      - condition: state
        entity_id: binary_sensor.windows_open
        state: "off"
    action:
      - service: climate.set_temperature
        target:
          entity_id: climate.wohnzimmer
        data:
          temperature: 22
      - service: conversation.process
        data:
          agent_id: conversation.freellm_chat
          text: >
            Die Klimaanlage wurde automatisch aktiviert.
            Außentemperatur: {{ states('sensor.outdoor_temperature') }}°C
            Innentemperatur: {{ states('sensor.indoor_temperature') }}°C
            Zieltemperatur: 22°C
            Fenster: alle geschlossen
            
            Erstelle eine kurze, freundliche Benachrichtigung die erklärt
            warum die Klimaanlage eingeschaltet wurde.
        response_variable: climate_info
      - service: notify.mobile_app_phone
        data:
          title: "❄️ Klimaanlage aktiviert"
          message: "{{ climate_info.response.speech.plain.speech }}"
